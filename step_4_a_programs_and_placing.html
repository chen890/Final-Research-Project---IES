    groupedvalues=df.groupby('label').sum().reset_index()
    pal = sns.color_palette(sns.color_palette("RdBu_r",sample(range(1,10),1)[0]+3 ), len(groupedvalues))

    for index, row in groupedvalues.iterrows():
